chartNodesData = {
  const dataMap = new Map();
  // We use filteredLinePubs so the bubbles react to the platform checkboxes!
  filteredLinePubs.forEach(d => {
    const sf = d.Subfield || "Unknown";
    const tp = d["Primary Topic"] || "Unknown";
    
    // Initialize Subfield
    if (!dataMap.has(sf)) dataMap.set(sf, { id: sf, type: "Subfield", count: 0, children: new Map() });
    const sfNode = dataMap.get(sf);
    sfNode.count += 1;
    
    // Initialize Topic inside Subfield
    if (!sfNode.children.has(tp)) sfNode.children.set(tp, { id: tp, parentId: sf, type: "Topic", count: 0 });
    sfNode.children.get(tp).count += 1;
  });

  const subfields = [];
  const topics = [];
  dataMap.forEach(sf => {
    subfields.push({ id: sf.id, type: "Subfield", count: sf.count });
    sf.children.forEach(tp => {
      topics.push({ id: tp.id, parentId: sf.id, type: "Topic", count: tp.count });
    });
  });
  return { subfields, topics };
}

viewof activeBubbleFilters = {
  const width = 1400;
  const height = 500;
  
  const svg = d3.create("svg")
    .attr("viewBox", [0, 0, width, height])
    .style("width", "100%")
    .style("height", "800px")
    .attr("class", "cursor-crosshair");

  // Background rect to catch double clicks for resetting
  svg.append("rect")
     .attr("width", width)
     .attr("height", height)
     .attr("fill", "transparent")
     .on("dblclick", resetChart);

  const maxCount = d3.max(chartNodesData.subfields, d => d.count) || 1;
  const rScale = d3.scaleSqrt().domain([0, maxCount]).range([15, 85]); 

  // NEW: Distinct color scale for topics based on their parent Subfield
  const colorScale = d3.scaleOrdinal(d3.schemeCategory10);

  function truncate(str, r) {
    const maxLen = Math.max(5, Math.floor(r / 3.5)); 
    if (str.length > maxLen) return str.substring(0, maxLen) + "..";
    return str;
  }

  // Initial load: Only Subfields
  let nodes = chartNodesData.subfields.map(d => ({
     ...d,
     r: rScale(d.count),
     x: width/2 + (Math.random()-0.5)*100,
     y: height/2 + (Math.random()-0.5)*100
  }));

  let focusedNodes = new Set();
  svg.node().value = []; // This array is what the database table will listen to

  const simulation = d3.forceSimulation(nodes)
    .force("x", d3.forceX(width/2).strength(0.04))
    .force("y", d3.forceY(height/2).strength(0.04))
    .force("collide", d3.forceCollide(d => d.r + 3).iterations(4));

  const nodeGroup = svg.append("g");
  let nodeElements = nodeGroup.selectAll("g");

  // Pushes changes down to the table
  function dispatchChange() {
    svg.node().value = Array.from(focusedNodes);
    svg.node().dispatchEvent(new CustomEvent("input"));
  }

function resetChart() {
     focusedNodes.clear();
     dispatchChange();
     // Bring all subfields back, spawning near the center
     nodes = chartNodesData.subfields.map(d => ({
        ...d,
        r: rScale(d.count),
        x: width/2 + (Math.random()-0.5)*150,
        y: height/2 + (Math.random()-0.5)*150
     }));
     update();
  }

  function onClick(event, d) {
    if (event.defaultPrevented) return; // Prevent triggering if they were just dragging

    // Burst Interaction for Subfields
    if (d.type === "Subfield") {
       nodes = nodes.filter(n => n.id !== d.id); // Remove the clicked subfield
       
       // Spawning the children in a geometric ring to prevent "Collision Explosions"
       const children = chartNodesData.topics.filter(t => t.parentId === d.id).map((t, i, arr) => {
         const angle = (i / arr.length) * Math.PI * 2; // Evenly space them around a circle
         const spawnRadius = d.r * 0.5; // Start them halfway out from the parent's center
         
         return {
           ...t,
           r: Math.max(12, rScale(t.count) * 0.8), 
           x: d.x + Math.cos(angle) * spawnRadius, // Trigonometry for X ring position
           y: d.y + Math.sin(angle) * spawnRadius, // Trigonometry for Y ring position
           vx: Math.cos(angle) * 3, // Give them a gentle, controlled push outward
           vy: Math.sin(angle) * 3
         };
       });
       
       nodes.push(...children);
       
       // Immediately add all daughter topics to the focused (active) list
       children.forEach(c => focusedNodes.add(c.id));
       
       dispatchChange();
       update();
    } 
    // Select/Toggle Interaction for Topics
    else {
       if (focusedNodes.has(d.id)) focusedNodes.delete(d.id);
       else focusedNodes.add(d.id);
       dispatchChange();
       updateStyles();
    }
  }

  function updateStyles() {
    nodeElements.selectAll("circle")
      .transition().duration(300)
      .attr("fill", d => {
         if (d.type === "Subfield") {
             if (focusedNodes.size === 0) return "#0077c8"; // All subfields start Bruker Blue
             return focusedNodes.has(d.id) ? "#0077c8" : "#cbd5e1"; // Grey out unclicked subfields
         } else {
             const topicColor = colorScale(d.parentId); // Distinct color per subfield family
             if (focusedNodes.size === 0) return topicColor; 
             return focusedNodes.has(d.id) ? topicColor : "#cbd5e1"; // Grey out unclicked topics
         }
      });
      
    nodeElements.selectAll("text")
      .attr("fill", d => focusedNodes.size > 0 && !focusedNodes.has(d.id) ? "#64748b" : "#ffffff");
  }

  function update() {
    nodeElements = nodeElements.data(nodes, d => d.id);
    nodeElements.exit().transition().duration(300).attr("transform", "scale(0)").remove();

    const enter = nodeElements.enter().append("g")
      .attr("transform", d => `translate(${d.x},${d.y})`)
      .style("cursor", "pointer")
      .on("click", onClick)
      .call(d3.drag().on("start", dragstarted).on("drag", dragged).on("end", dragended));

    enter.append("circle")
      .attr("r", d => d.r)
      .attr("stroke", "#fff")
      .attr("stroke-width", 2)
      .attr("fill", d => d.type === "Subfield" ? "#0077c8" : colorScale(d.parentId));

    enter.append("text")
      .text(d => truncate(d.id, d.r))
      .attr("text-anchor", "middle")
      .attr("dy", ".3em")
      .attr("fill", "#ffffff")
      .attr("font-size", d => Math.min(14, Math.max(9, d.r / 3.5)) + "px")
      .attr("font-weight", "600")
      .style("pointer-events", "none");

    // Tooltip on hover
    enter.append("title").text(d => `${d.type}: ${d.id}\nPublications: ${d.count}`);

    nodeElements = enter.merge(nodeElements);
    updateStyles();

    simulation.nodes(nodes);
    simulation.alpha(1).restart();
  }

  simulation.on("tick", () => {
    nodeElements.attr("transform", d => {
      // Keep nodes softly constrained to the SVG canvas
      d.x = Math.max(d.r, Math.min(width - d.r, d.x));
      d.y = Math.max(d.r, Math.min(height - d.r, d.y));
      return `translate(${d.x},${d.y})`;
    });
  });

  function dragstarted(event, d) {
    if (!event.active) simulation.alphaTarget(0.3).restart();
    d.fx = d.x; d.fy = d.y;
  }
  function dragged(event, d) {
    d.fx = event.x; d.fy = event.y;
  }
  function dragended(event, d) {
    if (!event.active) simulation.alphaTarget(0);
    d.fx = null; d.fy = null;
  }

  update();
  return svg.node();
}

      // If array is empty (nothing selected), return true. Otherwise, check if Subfield or Topic matches.
      const matchBubble = activeBubbleFilters.length === 0 || 
                          activeBubbleFilters.includes(p.Subfield) || 
                          activeBubbleFilters.includes(p["Primary Topic"]);
      
      return matchSearch && matchBubble;
  });
  
  // Sort
  filtered.sort((a, b) => {
    let valA = a[sortBy];
    let valB = b[sortBy];
    
    // Numeric fallback
    if (sortBy === "Year" || sortBy === "Cited By") {
      valA = Number(valA) || 0;
      valB = Number(valB) || 0;
      return sortOrder === "Desc" ? valB - valA : valA - valB;
    } else {
      // String compare
      valA = String(valA || "").toLowerCase();
      valB = String(valB || "").toLowerCase();
      if (valA < valB) return sortOrder === "Desc" ? 1 : -1;
      if (valA > valB) return sortOrder === "Desc" ? -1 : 1;
      return 0;
    }
  });
  
  return filtered;
}
